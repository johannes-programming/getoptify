Overview

This project allows getopt parsing followed by deparsing, e.g. for allowing abbreviations with click or to enforce parser simplicity.

def command(*_args, **_kwargs):
    return functools.partial(decorator, *_args, **_kwargs)

def decorator(old, /, *_args, **_kwargs):
    @functools.wraps(old)
    def new(args=None):
        args = process(args, *_args, **_kwargs)
        return old(args)

    return new
